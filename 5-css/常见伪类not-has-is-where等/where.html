/* Selects any paragraph inside a header, main
   or footer element that is being hovered */

    :where(header, main, footer) p:hover {
      color: red;
      cursor: pointer;
    }

    /* The above is equivalent to the following */

    header p:hover,
    main p:hover,
    footer p:hover {
      color: red;
      cursor: pointer;
    }
